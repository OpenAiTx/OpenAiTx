from pytorch_accelerated import Trainer

class MNISTModel(nn.Module):
    def __init__(self):
        super().__init__()
        self.main = nn.Sequential(

            nn.ReLU(),

def main():

        eval_dataset=validation_dataset,
        num_epochs=8,
        per_device_batch_size=32,
